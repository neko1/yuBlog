# 延迟链式调用

class PlayBoy {
  constructor(name) {
    this.name = name;
    setTimeout(() => this.next());
  }

  queue = [];

  next() {
    const fn = this.queue.shift();
    fn && fn();
  }

  sayHi() {
    this.queue.push(() => {
      console.log(`大家好我是${this.name}`);
      this.next();
    });
    return this;
  }

  sleep(seed) {
    this.queue.push(() => {
      console.log(`${seed / 1000}s 之后`);
      setTimeout(() => {
        this.next();
      }, seed);
    });
    return this;
  }

  play(game) {
    this.queue.push(() => {
      console.log(`我在玩${game}`);
      this.next();
    });
    return this;
  }
}
